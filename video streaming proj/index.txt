<script>

function searchVideos() {
  let input = document.getElementById("search").value.toLowerCase();
  let videos = document.getElementsByClassName("video-box");

  for (let i = 0; i < videos.length; i++) {
    let title = videos[i].innerText.toLowerCase();
    videos[i].style.display = title.includes(input) ? "block" : "none";
  }
}

function filterVideos(category) {
  let videos = document.getElementsByClassName("video-box");

  for (let i = 0; i < videos.length; i++) {
    if (category === "all") {
      videos[i].style.display = "block";
    } else {
      videos[i].style.display = videos[i].classList.contains(category) ? "block" : "none";
    }
  }
}
